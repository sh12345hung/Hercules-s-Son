	public static void CheckOut() 
	{
		try{
			Jsoup.connect(web).execute(); // connect to web
		}
		catch (HttpStatusException  e){   // if web cant connect => pr

			if(e.getStatusCode() == 400){
				System.out.println("Bad Request");
			}
			if(e.getStatusCode() == 401){
				System.out.println("Unauthorized");
				
			}
			if(e.getStatusCode()== 403){
				System.out.println("Forbidden");
			}
			if(e.getStatusCode()== 404){
				System.out.println("Not Found");
			}
			if(e.getStatusCode()== 408){
				System.out.println("Request Timeout");
			}
			if(e.getStatusCode()== 500){
				System.out.println("Internal Server Error");
			}
			if(e.getStatusCode()== 502){
				System.out.println("Bad Gateway");
			}
			if(e.getStatusCode()== 504){
				System.out.println("Gateway Timeout");
			}
		}		
		catch (IOException e)
		{
			
		}
	}
